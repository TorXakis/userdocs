TorXakis Grammar¶

Grammar: TorXakis model

  1   // Define TorXakis grammar
  2   grammar TorXakisLanguageDefinition;
  3  
  4   // grammar rules
  5   model                   :       (   typeDefs
  6                                   |   funcDefs
  7                                   |   constDefs
  8                                   |   procDefs
  9                                   |   stautDefs
 10                                   |   channelDefs
 11                                   |   modelDef
 12                                   |   purpDef
 13                                   |   mapperDef
 14                                   |   cnectDef
 15                                   )* EOF;
 16  
 17   typeDefs                :       'TYPEDEF' typeDef (';' typeDef)* 'ENDDEF';
 18   typeDef                 :       typeName '::=' neConstructorList;
 19  
 20   funcDefs                :       'FUNCDEF' funcDef (';' funcDef)* 'ENDDEF';
 21   funcDef                 :       funcName '(' ( neVarsDeclarationList ) ')' '::'
 22                                   typeName '::=' valExpr;
 23  
 24   constDefs               :       'CONSTDEF' constDef (';' constDef)* 'ENDDEF';
 25   constDef                :       constName '::' typeName '::=' valExpr;
 26  
 27   procDefs                :       'PROCDEF' procDef (';' procDef )* 'ENDDEF';
 28   procDef                 :       procName '[' ( neChannelsDeclList)? ']'
 29                                   '(' (neVarDeclList)? ')' (exitDecl)?
 30                                   '::=' processBehaviour;
 31  
 32   stautDefs               :       'STAUTDEF' stautDef (';' stautDef )* 'ENDDEF';
 33   stautDef                :       stautName '[' ( neChannelsDeclList)? ']'
 34                                   '(' (neVarDeclList)? ')' (exitDecl)?
 35                                   '::=' stautItems;
 36  
 37   channelDefs             :       'CHANDEF' channelDef 'ENDDEF';
 38   channelDef              :       channelDefName '::=' neChannelsDeclList;
 39  
 40  
 41   modelDef                :       'MODELDEF' modelName '::='
 42                                       'CHAN' 'IN'  ( neChannelNameList )?
 43                                       'CHAN' 'OUT' ( neChannelNameList )?
 44                                       'BEHAVIOUR' processBehaviour
 45                                   'ENDDEF';
 46  
 47   purpDef                 :       'PURPDEF' purpName '::='
 48                                       'CHAN' 'IN'  ( neChannelNameList )?
 49                                       'CHAN' 'OUT' ( neChannelNameList )?
 50                                       neTestGoals
 51                                   'ENDDEF';
 52  
 53   mapperDef               :       'MAPPERDEF' mapperName '::='
 54                                       'CHAN' 'IN'  ( neChannelsDeclList)?
 55                                       'CHAN' 'OUT' ( neChannelsDeclList )?
 56                                       'BEHAVIOUR' processBehaviour
 57                                   'ENDDEF';
 58  
 59   cnectDef                :       'CNECTDEF' cnectName '::='
 60                                       ( 'CLIENTSOCK' | 'SERVERSOCK' )
 61                                       connectionItem*
 62                                   'ENDDEF';
 63  
 64  
 65   exitDecl                :       'EXIT' ('::' neTypeNameList)?;
 66  
 67  
 68   stautItems              :       (    stateItem
 69                                   |    varItem
 70                                   |    initItem
 71                                   |    transItem
 72                                   )*;
 73   stateItem               :       'STATE' neIdNameList;
 74   neIdNameList            :       idName (',' idName)*;
 75   varItem                 :       'VAR' (neVarsDeclarationList)?;
 76   initItem                :       'INIT' idName (updateList)?;
 77   transItem               :       'TRANS' transition (';' transition)*;
 78  
 79   transition              :       idName '->' conditionalCommunications
 80                                   (updateList)? '->' idName;
 81  
 82   updateList              :       '{' update (';' update )* '}';
 83   update                  :       varName ':=' valExpr;
 84  
 85   neConstructorList       :       constructor ( '|' constructor )*;
 86   constructor             :       constructorName  ( '{' neFieldList '}' )?;
 87  
 88   neFieldList             :       fields ( ';' fields )*;
 89   fields                  :       neFieldNameList '::' typeName;
 90   neFieldNameList         :       fieldName ( ',' fieldName )*;
 91  
 92  
 93  
 94   neTestGoals             :       testGoal (testGoal)*;
 95   testGoal                :       'GOAL' idName '::=' processBehaviour;
 96  
 97   connectionItem          :       (    connectionOut
 98                                   |    connectionIn
 99                                   |    encoding
100                                   |    decoding
101                                   );
102   connectionOut           :       'CHAN' 'OUT' channelsDecl
103                                   'HOST' hostName 'PORT' portNumber;
104   connectionIn            :       'CHAN' 'IN' channelsDecl
105                                   'HOST' hostName 'PORT' portNumber;
106   encoding                :       'ENCODE' communication '->' channelOffer;
107   decoding                :       'DECODE' communication '<-' channelOffer;
108  
109  
110  
111   neVarsDeclarationList   :       varsDeclaration (';' varsDeclaration )*;
112   varsDeclaration         :       neVarNameList '::' typeName;
113   neVarDeclList           :       varsDecl (';' varsDecl)*;
114   varsDecl                :       neVarNameList '::' typeName;
115   neVarNameList           :       varName (',' varName)*;
116  
117   neChannelsDeclList      :       channelsDecl ( ';' channelsDecl )*;
118   channelsDecl            :       neChannelNameList ( '::' neTypeNameList)?;
119   neChannelNameList       :       channelName (',' channelName)*;
120  
121   neTypeNameList          :       typeName ('#' typeName)*;
122  
123  
124  
125   processBehaviour        :       processBehaviourLevel1;
126  
127   processBehaviourLevel1  :       processBehaviourLevel2
128                                   ( (  '>>>' processBehaviourLevel2 )
129                                   | (  '>>>' 'ACCEPT' ( '?' (varDecls+=varDecl |
130                                        varName) | '!' valExpr )* 'IN'
131                                        processBehaviourLevel2 'NI' )
132                                   | (  '[>>' processBehaviourLevel2 )
133                                   | (  '[><' processBehaviourLevel2 )
134                                   )*;
135  
136   processBehaviourLevel2:         processBehaviourLevel3
137                                   ( ( '||' processBehaviourLevel3 )
138                                   | ( '|||' processBehaviourLevel3 )
139                                   | ( synchronizedChannels processBehaviourLevel3 )
140                                   )*;
141  
142   processBehaviourLevel3:         processBehaviourLevel4
143                                   ( (  '##' processBehaviourLevel4 )
144                                   )*;
145  
146   processBehaviourLevel4:         processBehaviourGuarded
147                                   | processBehaviourStop
148                                   | processBehaviourSequence
149                                   | procCall
150                                   | processBehaviourLet
151                                   | processBehaviourHide
152                                   | processBehaviourBracket;
153  
154   procCall                :       procName '[' (neChannelNameList)? ']'
155                                   '(' (neValExprList)? ')';
156  
157   neValExprList           :       valExpr ( ',' valExpr )*;
158  
159   processBehaviourBracket :       '(' processBehaviourLevel1 ')';
160   processBehaviourHide    :       'HIDE' '[' (neChannelsDeclList)? ']' 'IN'
161                                    processBehaviourLevel1 'NI';
162   processBehaviourLet     :       'LET' assignment ( ';' assignment )* 'IN'
163                                   processBehaviourLevel1 'NI';
164   processBehaviourSequence:       conditionalCommunications
165                                   ( '>->' processBehaviourLevel4 )?;
166   processBehaviourGuarded :       condition '=>>' processBehaviourLevel4;
167   processBehaviourStop    :       'STOP';
168  
169   synchronizedChannels    :       '|[' neChannelNameList ']|';
170  
171  
172   conditionalCommunications:      communications (condition)?;
173  
174   communications          :       communication ( '|' communication )*;
175  
176   communication           :       ( (channelName | 'EXIT' ) channelOffer* );
177  
178   channelOffer            :       '!' valExpr | '?' (varDecls+=varDecl | varName);
179  
180   condition               :       '[[' valExpr ']]';
181  
182   assignment              :       ( varDecl | varName ) '=' valExpr;
183  
184   varDecl                 :       varName '::' typeName;
185  
186   valExpr                 :       valExpr1;
187  
188   valExpr1                :       valExpr2
189                                   ( (  OPERATOR valExpr2 )
190                                   | (  '::' typeName )
191                                   )*
192                                   | valExprLet
193                                   | valExprIte;
194  
195  
196   valExpr2                :       smallIdName // Temporarily solution for the
197                                   // conflict in the next two
198                                   //        valExprConst
199                                   //    |    valExprVar
200                                   |    valExprUnaryOperator
201                                   |    valExprFunctionCall
202                                   |    valExprContructorCall
203                                   |    valExprInteger
204                                   |    valExprString
205                                   |    valExprRegex
206                                   |    valExprBracket
207                                   |    valExprError;
208  
209   valExprUnaryOperator    :       OPERATOR valExpr2;
210  
211   smallIdName             :       SMALLID;
212  
213   valExprError            :       'ERROR' STRING;
214   valExprIte              :       'IF' valExpr1
215                                   'THEN' valExpr1 'ELSE' valExpr1 'FI';
216   valExprLet              :       'LET' assignment (';' assignment)*
217                                   'IN' valExpr1 'NI';
218   valExprBracket          :       '(' valExpr ')';
219   valExprRegex            :       'REGEX' '(' STRING ')';
220   valExprString           :       STRING;
221   valExprInteger          :       INT;
222   valExprContructorCall   :       constructorName ( '(' neValExprList ')' )?;
223   valExprFunctionCall     :       funcName '(' ( neValExprList )? ')';
224   valExprVar              :       varName;
225   valExprConst            :       constName;
226  
227   // identifier types
228   portNumber              :       INT;
229  
230   hostName                :       STRING;
231  
232   idName                  :       SMALLID | CAPSID;
233  
234   modelName               :       CAPSID;
235   purpName                :       CAPSID;
236   mapperName              :       CAPSID;
237   cnectName               :       CAPSID;
238   stautName               :       CAPSID;
239   channelDefName          :       CAPSID;
240   constructorName         :       CAPSID;
241   typeName                :       CAPSID;
242   channelName             :       CAPSID;
243  
244   fieldName               :       SMALLID;
245   procName                :       SMALLID;
246   funcName                :       SMALLID;
247   constName               :       SMALLID;
248   varName                 :       SMALLID;
249  
250   // lexer rules; implicit lexer rules are in grammar rules above
251   OPERATOR        : ( '=' | '+' | '-' | '*' | '^' | '/' | '\\' | '<' | '>' |
252                       '@' | '|' | '&' | '%' )+ ;
253   CAPSID          : 'A'..'Z' ( 'A'..'Z' | 'a'..'z' | '0'..'9' | '_')*;
254   SMALLID         : 'a'..'z' ( 'A'..'Z' | 'a'..'z' | '0'..'9' | '_')*;
255  
256  
257   INT             : DIGIT+ ;
258   fragment DIGIT : '0'..'9' ;
259  
260   STRING : '"' .*? '"' ; // match anything in "..."
261  
262  
263   WS : [ \t\n\r]+ -> channel(1) ;
264  
265   LINE_COMMENT    : '--' .*? '\n' -> channel(2) ;
266   BLOCK_COMMENT   : '{-' .*? '-}' -> channel(2) ;
